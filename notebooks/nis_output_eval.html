data_dict = { 'radar': {}, 'lidar': {}} # A container for the radar/lidar NIS (for each of the tracked car)

for data_file in glob('../build/run-6-2-13:11:32/*.csv'):
    car, sensor = data_file.split(os.sep)[-1].rstrip('.csv').split('_')
    if sensor in data_dict:
        data_dict[sensor][car] = pd.read_csv(data_file).values
    else:
        data_dict[sensor] = {car: pd.read_csv(data_file).valus}
